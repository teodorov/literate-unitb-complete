\begin{calculation}
	f.x \le f.y

	f.x \uparrow f.y \,=\, f.y

	f.(x \uparrow y) \,=\, f.y
\hint{\follows}{ Leibniz }
	x \uparrow y \,=\, y
\hint{=}{ $\le$ from $=$ and $\uparrow$ }
	x \le y
\end{calculation}
