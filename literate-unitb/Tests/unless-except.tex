\variable{p : \Int } \\
\end{align*}

\newevent{evt0}{event\_0}
\newevent{evt1}{event\_1}

\begin{align*}
\evassignment{evt0}{m0:act0}{ p' = p + 1 }

\end{align*}

\indices{evt0}{ i : \Int }
\indices{evt1}{ i : \Int }

\end{align*}

% \safetyB{saf0}{evt0}{ k \le f.i }{ \false }
\\ \safety{saf1}{ k \le f.i }{ \false }
