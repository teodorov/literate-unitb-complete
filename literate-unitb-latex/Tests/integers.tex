\begin{proof}{evt/INV/inv0}
	\begin{calculation}
		(n')^3
	\hint{=}{ \ref{a0} }
		(n+1)^3
	\hint{=}{ arithmetic }
		n^3 + 3 \cdot n^2 + 3 \cdot n + 1
	\hint{=}{ \ref{inv0} }
		a + 3 \cdot n^2 + 3 \cdot n + 1
	\hint{=}{ we add a variable $b$: \ref{inv1}, see below }

	\hint{=}{ \ref{a1} }
		a'
	\end{calculation}
\end{proof}
